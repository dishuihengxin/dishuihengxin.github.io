        """æ ¹æ®æ…¢æŸ¥è¯¢å»ºè®®ç´¢å¼•"""

        for query in slow_queries:
            suggestion = {
                'collection': query.get('ns', '').split('.')[-1],

                if index_fields:
                    suggestion['suggested_indexes'].append({
                        'fields': index_fields,
                        'type': 'compound'
